    }

        int StreetId PK
        nvarchar Name

    }

    Faculties {
        int FacultyId PK
        nvarchar ShortName
        nvarchar Name
    }

    Groups {
        int GroupId PK
        nvarchar GroupName
        int FacultyId FK
    }

    Students {
        int StudentId PK
        nvarchar Surname
        nvarchar Name
        nvarchar MiddleName
        int StreetId FK
        smallint HouseNumber
        smallint ApartmentNumber
        bigint PhoneNumber
        date BirthDate
        smallint AdmissionYear
        int GroupId FK
        nvarchar Note
    }

    Purposes {
        int PurposeId PK
        nvarchar Name
    }

    Payments {
        int PaymentId PK
        int StudentId FK

        decimal Amount
        int PurposeId FK
